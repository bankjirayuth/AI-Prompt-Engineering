Good Prompt

Input (The 6 Pillars)

Persona: Senior Node.js Developer

Context: ระบบตะกร้าสินค้า E-commerce

Task: ฟังก์ชัน calculateFinalPrice(price, memberType, couponCode)

Constraints:
  • Gold ลด 10%, Silver ลด 5%
  • คูปอง 'WELCOME' ลด 50 บาท (คิดทีหลัง)
  • ห้ามราคาติดลบ, Throw Error ถ้า Input ผิด
  • ใช้ Clean Code + JSDoc

Output: JavaScript Code Only

calculateFinalPrice.js ✅ 
/**
 * @param {number} price - ราคาสินค้า
 * @param {'gold'|'silver'|'normal'} memberType
 * @param {string} [couponCode] - รหัสคูปอง
 * @returns {number} ราคาสุทธิ
 */
const calculateFinalPrice = (price, memberType, couponCode) => {
  if (typeof price !== 'number' || price < 0)
    throw new Error('Invalid price');

  const discountRates = { gold: 0.10, silver: 0.05 };
  let finalPrice = price * (1 - (discountRates[memberType] || 0));

  if (couponCode === 'WELCOME') finalPrice -= 50;
  return Math.max(0, finalPrice);
};
